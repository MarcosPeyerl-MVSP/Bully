CREATE TABLE UserRespostas (
    id_resp INT PRIMARY KEY AUTO_INCREMENT,
    data_resp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    somaTotal_resp INT NOT NULL,
    perfil_resp ENUM('vítima', 'agressor', 'espectador') NOT NULL
);

CREATE TABLE Resposta (
    id_opcao INT PRIMARY KEY AUTO_INCREMENT,
    id_pergunta INT,
    texto_opcao TEXT NOT NULL,
    pontuacao INT NOT NULL,
    FOREIGN KEY (id_pergunta) REFERENCES Perguntas(id_pergunta)
);

CREATE TABLE Perguntas (
    id_perg INT PRIMARY KEY AUTO_INCREMENT,
    texto_perg TEXT NOT NULL,
    ordem_perg INT NOT NULL
);
