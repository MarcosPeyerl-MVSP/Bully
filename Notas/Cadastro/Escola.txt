-- Tabela Escola
CREATE TABLE escola (
    id_escola INT PRIMARY KEY AUTO_INCREMENT,
    nome_escola VARCHAR(150) NOT NULL,
    categoria_escola ENUM('publica', 'privada') NOT NULL,
    uf_escola CHAR(2) NOT NULL,
    bairro_escola VARCHAR(100) NOT NULL,
);
